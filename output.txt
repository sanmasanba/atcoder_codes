"class BIT:",
    "\tdef __init__(self, n: int):",
        "\t\tself.n = n+1",
        "\t\tself.L = [0]*self.n",
    "\t",
    "\tdef add(self, idx: int, val: int):",
        "\t\t\"\"\"１点加算",
"",
        "\t\tArgs:",
            "\t\t\tidx (int): 1-indexed",
            "\t\t\tval (int): 加算する値",
        "\t\t\"\"\"",
        "\t\twhile idx < self.n:",
            "\t\t\tself.L[idx] += val",
            "\t\t\tidx += (idx & -idx)",
    "\t",
    "\tdef sum(self, idx: int) -> Any:",
        "\t\t\"\"\"区間取得",
"",
        "\t\tArgs:",
            "\t\t\tidx (int): 1-indexed",
"",
        "\t\tReturns:",
            "\t\t\tAny: 区間和",
        "\t\t\"\"\"",
        "\t\tres = 0",
        "\t\twhile idx > 0:",
            "\t\t\tres += self.L[idx]",
            "\t\t\tidx -= (idx & -idx)",
        "\t\treturn res",
